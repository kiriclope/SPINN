  def get_times():
       config = safe_load(open("/home/leon/models/lif_cpp/conf/config_EI.yml", "r"))

       times = np.arange(config['T_STEADY'], config['DURATION'], config['T_WINDOW'])
       times -= config['T_STEADY']
       times /= 1000.0
       
       return times
